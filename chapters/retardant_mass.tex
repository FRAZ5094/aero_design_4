\chapter{Mass of retardant}

Density of retardant = 9 lb/gallon \\ 
1 lb = 0.45kg

\section{3000 gallons}
\begin{equation}
\begin{split}
  m & = \rho * V \\
  m_{3000} & = 9 * 3000 \\
  & = 27000 \ lbs \\
  & = 12150 \ kg 
\end{split}
\end{equation}

\section{5000 gallons}
\begin{equation}
\begin{split}
  m & = \rho * V \\
  m_{5000} & = 9 * 5000 \\
  & = 45000 \ lbs \\
  & = 20250 \ kg 
\end{split}
\end{equation}
